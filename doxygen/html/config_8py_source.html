129  def _yaml_parse_where(self, data):
130  """Private method to parse 'where' data.
131 
132  Args:
133  data (dict): The YAML data.
134 
135  Returns:
136  None
137  """
138  tables = self.configconfig.get("tables", [])
139  # No tables specified, extract all tables
140  if not tables:
141  tables = ["nodes", "ways_line", "ways_poly"]
142 
143  for table in tables:
144  self.configconfig["where"][table] = []
145  where_entries = data.get("where", {}).get("tags", [])
146 
147  for entry in where_entries:
148  if entry == "op":
149  # Skip if already in correct {'op': 'or'} format
150  # Necessary?
151  continue
152 
153  # If no join passed, default to join_or
154  if not any(key.startswith("join_") for key in entry.keys()):
155  entry = {"join_or": [entry]}
156 
157  # Extract key / values
158  [[key, value]] = entry.items()
159 
160  # Get the operation type after join_
161  op = key.split("_")[1]
162 
163  for tags in value:
164  for tag, selector in tags.items():
165  if selector is True:
166  # yes is required in osm tag query instead of true
167  selector = "yes"
168  if selector is None:
169  # empty instead of None (equivalent to not null)
170  newtag = {tag: [], "op": op}
171  else:
172  newtag = {tag: [selector], "op": op}
173  self.configconfig["where"][table].append(newtag)

226  # Helper function to convert geometry names
227  def convert_geometry(geom_type):
228  if geom_type == "point":
229  return "nodes"
230  elif geom_type == "line":
231  return "ways_line"
232  elif geom_type == "polygon":
233  return "ways_poly"
234  return geom_type
235 
236  # Extract geometry
237  if geom_dict := data.get("geometry"):
238  self.geometrygeometry = shape(geom_dict)
239 
240  # Iterate through each key-value pair in the flattened dictionary
241  for key, value in flatdict.FlatDict(data).items():
242  keys = key.split(":")
243  # Skip the keys related to geometry
244  if key.startswith("geometry"):
245  continue
246  # If it's a top-level key, directly update self.config
247  if len(keys) == 1:
248  self.configconfig[key] = value
249  continue
250 
251  # Extract meaningful parts from the key
252  section, subsection = keys[:2]
253  geom_type = keys[2] if len(keys) > 2 else None
254  tag_type = keys[3] if len(keys) > 3 else None
255  tag_name = keys[4] if len(keys) > 4 else None
256 
257  # Convert geometry type to meaningful names
258  geom_type = convert_geometry(geom_type)
259 
260  if subsection == "attributes":
261  # For attributes, update select fields and tables
262  for attribute_name in value:
263  # FIXME needs a refactor to handle all_geometry correctly
264  if geom_type == "all_geometry":
265  for geometry_type in ["nodes", "ways_line", "ways_poly"]:
266  self.configconfig["select"][geometry_type].append(
267  {attribute_name: {}}
268  )
269  self.configconfig["tables"].append(geometry_type)
270  else:
271  self.configconfig["select"][geom_type].append({attribute_name: {}})
272  self.configconfig["tables"].append(geom_type)
273  elif subsection == "tags":
274  # For tags, update where fields
275  option = tag_type[5:] if tag_type else None
276  new_tag = {tag_name: value, "op": option}
277  if geom_type == "all_geometry":
278  for geometry_type in ["nodes", "ways_line", "ways_poly"]:
279  self.configconfig["where"][geometry_type].append(new_tag)
280  else:
281  self.configconfig["where"][geom_type].append(new_tag)
